n:
 memint 1 ; Определяем целое число n
decr_value:
 memint 1 ; Определеяем значение для декремента
res:
 memint 1 ; Здесь кладём временный результат
start:
 ini n    ; Вводим число n
 push n   ; Кладём в стек
 ini decr_value ; Вводим счётчик
cycle:
 push decr_value ; Кладём в стек счётчик
 isub            ; Вычитаем
 peek res        ; Сохраняем по адресу
 outi res        ; Выводим
 jnz cycle       ; Пока вычисленное значение не равно нулю
 end start       ; Выведутся числа от n - 1 до 0
